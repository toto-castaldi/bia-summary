   ```typescript
   export interface AppConfig {
     anthropicKey: string;
     cloudConvertKey: string;
   }

   export interface PipelineOptions {
     inputPath: string;
     dryRun: boolean;
   }

   export interface AnalysisResult {
     markdown: string;
     stopReason: string;
     model: string;
     inputTokens: number;
     outputTokens: number;
   }
   ```

2. Create `src/config.ts` with two exported functions:

   **`validateEnv(): AppConfig`** -- Per D-06, D-07, D-08:
   - Import `z` from `"zod"` (Zod 4 default export)
   - Define `envSchema = z.object({ ANTHROPIC_API_KEY: z.string().min(1, "ANTHROPIC_API_KEY is required"), CLOUDCONVERT_API_KEY: z.string().min(1, "CLOUDCONVERT_API_KEY is required") })`
   - Call `envSchema.safeParse(process.env)`
   - On failure: write to stderr listing exactly which keys are missing, with message "Set them in a .env file (working directory) or as environment variables.", then `process.exit(1)`
   - On success: return `{ anthropicKey: result.data.ANTHROPIC_API_KEY, cloudConvertKey: result.data.CLOUDCONVERT_API_KEY }`
   - Note: Both keys validated even in Phase 1 (D-08) for forward consistency

   **`loadPrompt(pdfPath: string): Promise<string>`** -- Per D-01, D-03:
   - Import `readFile`, `access` from `"node:fs/promises"` and `path` from `"node:path"`
   - Resolve pdfPath to absolute: `path.resolve(pdfPath)`
   - Compute promptPath: `path.join(path.dirname(resolvedPdfPath), "prompt.txt")`
   - Try `access(promptPath)`. On failure: write to stderr the exact message:
     ```
     Prompt file not found.
     Expected at: {promptPath}
     Place your prompt.txt file in the same directory as the input PDF.
     ```
     Then `process.exit(1)`
   - On success: return `readFile(promptPath, "utf-8")`

3. Run `npx tsc --noEmit` to verify both files compile without errors.
